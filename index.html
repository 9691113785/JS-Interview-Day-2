What is creation phase and execution phase?

    Creation Phase : The creation phase is the one in which the JS engine invokes a function but does not begin its execution
    Execution Phase : This is the stage where the actual code execution begins.

What are callbacks?

    
A callback is a function passed as an argument to another function. 
This technique allows a function to call another function.

example:
function alok (abhi){
    console.log('hello');
    abhi();
    console.log('mishra');
}
function abhi(){
    console.log('abhi');
}
alok(abhi)

Explain prototypes

    Every object in JavaScript has a built-in property, which is called its prototype. 
    The prototype is itself an object, so the prototype will have its own prototype, 
    making what's called a prototype chain.

    example:
    var Mob = function(name,id){
        this.name=name,
        this.id=id
        
        }
        Mob.prototype.abhi=function(){
            console.log(`${this.name} and ${this.id}`);
        }
        
        var sum = new Mob('alok', 5)
        var all = new Mob('aa', 8)
        
        console.log(sum.abhi());
        console.log(all.abhi());
        console.log(sum);

Give an example of inheritance using function constructor

    function Employee(name, age, gender, id) {
        this.name = name;
        this.age = age;
        this.gender = gender;
        this.id = id;
    };

    function Developer(name, age, gender, id, 
    specialization) {

        // Calling Employee constructor function
        Employee.call(this, name, age, gender, id);

        // Adding a new parameter
        this.specialization = specialization;
    }
    Employee('alok',5,'M',5)
    Developer('abhi',5,'M',5)
